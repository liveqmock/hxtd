关于日历组件，使用jquery选择器来批量或者单个绑定文本框

$(function(){
    jsUtil.datepicker("#datepicker");//#datepicker选择器
});

给文本框绑定显示的树菜单：
$(function(){
    jsUtil.easyTree.init(zNodes,chileNodeFun);
    $("#easyTree").click(function(){//给文本框绑定点击事件
        jsUtil.easyTree.show("#easyTree");//点击时触发
    });
});
